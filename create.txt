CREATE TABLE UYE(
ID INTEGER NOT NULL,
ADI VARCHAR(30),
SOYADI VARCHAR(30),
SEHIR VARCHAR(15),
PRIMARY KEY (ID)
);
CREATE TABLE ULKE (
ID INTEGER NOT NULL ,
ADI VARCHAR(30),
PRIMARY KEY (ID)
);
CREATE TABLE ELESTIRMEN(
ID INTEGER NOT NULL ,
ADI VARCHAR(30),
SOYADI VARCHAR(30),
PRIMARY KEY (ID)
);
CREATE TABLE ODUL (
ID INTEGER NOT NULL ,
ADI VARCHAR(30),
PRIMARY KEY (ID)
);
CREATE TABLE TURLER (
ID INTEGER NOT NULL ,
ADI VARCHAR(30),
PRIMARY KEY (ID)
);
CREATE TABLE YONETMEN (
ID INTEGER NOT NULL ,
ADI VARCHAR(30),
SOYADI VARCHAR(30),
CINSIYET VARCHAR(5),
ULKE INTEGER,
PRIMARY KEY (ID),
FOREIGN KEY (ULKE) REFERENCES ULKE(ID)
);
CREATE TABLE AKTOR (
ID INTEGER NOT NULL ,
ADI VARCHAR(30),
SOYADI VARCHAR(30),
CINSIYET VARCHAR(5),
ULKE INTEGER,
PRIMARY KEY (ID),
FOREIGN KEY (ULKE) REFERENCES ULKE(ID)
);
CREATE TABLE FILM (
ID INTEGER NOT NULL ,
ADI VARCHAR(30),
YAPIM_YILI INTEGER,
SURE INTEGER,
DIL VARCHAR(20),
ULKE INTEGER,
PRIMARY KEY (ID),
FOREIGN KEY (ULKE) REFERENCES ULKE(ID)
);
CREATE TABLE FILMYONETMENLERI (
FILMID INTEGER NOT NULL ,
YONETMENID INTEGER NOT NULL ,
FOREIGN KEY (FILMID) REFERENCES FILM(ID),
FOREIGN KEY (YONETMENID) REFERENCES YONETMEN(ID)
);
CREATE TABLE FILMAKTORLERI (
FILMID INTEGER NOT NULL ,
AKTORID INTEGER NOT NULL ,
ROL VARCHAR(20),
CONSTRAINT FILM_FILMAKTOR_FK FOREIGN KEY (FILMID) REFERENCES FILM(ID),
CONSTRAINT AKTOR_FILMAKTOR_FK FOREIGN KEY (AKTORID) REFERENCES AKTOR(ID)
);
CREATE TABLE FILMTURLERI (
FILMID INTEGER,
TURID INTEGER,
FOREIGN KEY (FILMID) REFERENCES FILM(ID),
FOREIGN KEY (TURID) REFERENCES TURLER(ID)
);
CREATE TABLE FILMODULLERI (
FILMID INTEGER,
ODULID INTEGER,
FOREIGN KEY (FILMID) REFERENCES FILM(ID),
FOREIGN KEY (ODULID) REFERENCES ODUL(ID)
);
CREATE TABLE FILMELESTIRILERI(
FILMID INTEGER,
PUAN INTEGER,
ELESTIRMENID INTEGER,
FOREIGN KEY (ELESTIRMENID) REFERENCES ELESTIRMEN(ID),
FOREIGN KEY (FILMID) REFERENCES FILM(ID)
);
CREATE TABLE ODUNCFILMLER(
UYEID INTEGER,
FILMID INTEGER,
ALINMATARIHI DATE,
IADDETARIHI DATE,
FOREIGN KEY (UYEID) REFERENCES UYE(ID),
FOREIGN KEY (FILMID) REFERENCES FILM(ID)
);
